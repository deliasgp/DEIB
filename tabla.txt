tabla <- function(df, ...) {
  library(dplyr)
  library(tidyverse)
  group_var <- quos(...)
  i=length(group_var)
  n=nrow(df)
  df.1<-df %>%
      group_by(!!!group_var) %>%
      summarise(t = n()) %>%
      mutate(p.t=t/n*100)
  
  j=as.numeric(length(df.1))
  i=as.numeric(i)
  u=as.numeric(i+1)
  x1=rep(0,i)
  x2=colSums(df.1[,u:j])
  x3=c(x1,x2)
  df.1=rbind.data.frame(x3,df.1)
}
